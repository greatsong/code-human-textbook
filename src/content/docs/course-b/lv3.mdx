### 프로젝트 1: 식물 성장 환경 분석

**연구 질문:** "교실 안 위치별로 온도와 빛의 양이 얼마나 다를까? 식물을 어디에 두면 좋을까?"

**가설:** "창가의 평균 조도는 교실 안쪽보다 3배 이상 높을 것이다."

**센서:** DHT20 온습도 (I2C, 0x38) + 빛 센서 (ADC, GP26) + OLED

**실험 설계:**

| 항목 | 내용 |
|------|------|
| 독립 변인 | 측정 위치 (창가 vs 안쪽) |
| 종속 변인 | 조도 (%), 온도 (°C) |
| 통제 변인 | 측정 시간대, 날씨, 커튼 상태 |
| 측정 기간 | 1시간 (위치별 30분) |
| 측정 간격 | 60초 |

#### 전체 코드

```python
from machine import I2C, Pin, ADC
import time

# ── 연구 설정 ──
RESEARCH_TITLE = "온도와 광량이 식물에 미치는 영향"
MEASURE_INTERVAL = 60  # 60초마다 측정
LOG_FILE = "plant_research.csv"

# ── 센서 설정 ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
dht_addr = 0x38
light = ADC(Pin(26))  # Grove A0 포트
time.sleep(0.1)

from ssd1306 import SSD1306_I2C
oled = SSD1306_I2C(128, 64, i2c)

def read_dht20():
    i2c.writeto(dht_addr, bytes([0xAC, 0x33, 0x00]))
    time.sleep(0.08)
    data = i2c.readfrom(dht_addr, 7)
    humi = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4)) / 1048576 * 100
    temp = (((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]) / 1048576 * 200 - 50
    return round(temp, 1), round(humi, 1)

# ── CSV 파일 준비 ──
import os
new_file = LOG_FILE not in os.listdir()
f = open(LOG_FILE, "a")
if new_file:
    f.write("측정번호,경과시간(분),온도(C),습도(%),광량(%)\n")
count = 0
start = time.ticks_ms()
temp_hist = []  # 온도 트렌드 (최대 40개)

# ── 장기 데이터 수집 루프 ──
while True:
    temp, humi = read_dht20()
    light_pct = round(light.read_u16() / 65535 * 100, 1)
    count += 1
    elapsed_min = time.ticks_diff(time.ticks_ms(), start) // 60000

    # 이력 저장
    temp_hist.append(int(temp))
    if len(temp_hist) > 40:
        temp_hist.pop(0)

    # OLED 연구 대시보드
    oled.fill(0)
    # 헤더 바
    oled.fill_rect(0, 0, 128, 12, 1)
    oled.text("Plant Lab", 28, 2, 0)
    oled.hline(0, 13, 128, 1)
    # 측정 정보
    oled.text(f"#{count}", 4, 16)
    oled.text(f"{elapsed_min}min", 80, 16)
    # 센서 값
    oled.text(f"T:{temp}C", 4, 28)
    oled.text(f"L:{light_pct}%", 72, 28)
    # 온도 트렌드 그래프
    oled.hline(0, 40, 128, 1)
    oled.text("Trend", 0, 42)
    if len(temp_hist) > 1:
        t_min = min(temp_hist)
        t_max = max(temp_hist)
        t_range = max(t_max - t_min, 1)
        for i in range(len(temp_hist)):
            h = int((temp_hist[i] - t_min) / t_range * 16)
            x = 44 + i * 2
            oled.vline(x, 62 - h, max(h, 1), 1)
    oled.show()

    # CSV 저장
    f.write(f"{count},{elapsed_min},{temp},{humi},{light_pct}\n")
    f.flush()

    print(f"#{count} | {elapsed_min}분 | T:{temp} H:{humi} L:{light_pct}%")
    time.sleep(MEASURE_INTERVAL)
```

```python
# PC에서 실행 (Pico가 아님!)
import matplotlib.pyplot as plt
import csv

# CSV 파일 읽기
data = {'time': [], 'temp': [], 'humi': [], 'light': []}
with open('plant_research.csv', 'r') as f:
    reader = csv.DictReader(f)
    for row in reader:
        data['time'].append(int(row['경과시간(분)']))
        data['temp'].append(float(row['온도(C)']))

ax2.plot(data['time'], data['light'], 'y-o', label='Light (%)')
ax2.set_xlabel('Time (min)')
ax2.set_ylabel('Light (%)')
ax2.legend()
ax2.grid(True)

plt.suptitle('Plant Growth Environment Analysis')
plt.tight_layout()
plt.savefig('plant_analysis.png')
plt.show()
```

```python
# PC 실시간 대시보드 — 식물 성장 환경 분석
# 사용법: pip install pyserial matplotlib
import serial
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
from matplotlib.widgets import Button
from collections import deque
import time, csv

PORT = '/dev/tty.usbmodem1101'  # 환경에 맞게 수정
BAUD = 115200
ser = serial.Serial(PORT, BAUD, timeout=1)
time.sleep(2)

MAX_POINTS = 200
times = deque(maxlen=MAX_POINTS)
temps = deque(maxlen=MAX_POINTS)
humis = deque(maxlen=MAX_POINTS)
lights = deque(maxlen=MAX_POINTS)

plt.style.use('dark_background')
fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(10, 7))
fig.suptitle('Plant Growth Environment', fontsize=14, color='cyan')

# CSV 내보내기 버튼
csv_file = 'plant_pc_log.csv'
def export_csv(event):
    with open(csv_file, 'w', newline='') as f:
        w = csv.writer(f)
        w.writerow(['time_min', 'temp', 'humi', 'light_pct'])
        for i in range(len(times)):
            w.writerow([f'{times[i]:.1f}', f'{temps[i]:.1f}',
                        f'{humis[i]:.1f}', f'{lights[i]:.1f}'])
    print(f'Saved {len(times)} rows to {csv_file}')
ax_btn = plt.axes([0.81, 0.01, 0.15, 0.04])
btn = Button(ax_btn, 'Export CSV', color='#333', hovercolor='#555')
btn.on_clicked(export_csv)

def update(frame):
    try:
        line = ser.readline().decode('utf-8').strip()
        if not line or not line.startswith('#'): return
        # 파싱: #1 | 0분 | T:24.5 H:55.2 L:72.3%
        parts = line.split('|')
        elapsed = float(parts[1].strip().replace('분', ''))
        vals = parts[2].strip().split()
        t = float(vals[0].split(':')[1])
        h = float(vals[1].split(':')[1])
        l = float(vals[2].split(':')[1].replace('%', ''))
        times.append(elapsed)
        temps.append(t)
        humis.append(h)
        lights.append(l)

        ax1.clear()
        ax1.set_title('Temperature & Humidity')
        ax1.set_ylabel('Temperature (C)', color='#ff6b6b')
        ax1.plot(list(times), list(temps), color='#ff6b6b', linewidth=2, label='Temp')
        ax1.tick_params(axis='y', labelcolor='#ff6b6b')
        ax1r = ax1.twinx()
        ax1r.set_ylabel('Humidity (%)', color='#4ecdc4')
        ax1r.plot(list(times), list(humis), color='#4ecdc4', linewidth=2, label='Humi')
        ax1r.tick_params(axis='y', labelcolor='#4ecdc4')
        ax1.set_xlabel('Time (min)')

        ax2.clear()
        ax2.set_title('Light Level')
        ax2.fill_between(list(times), list(lights), alpha=0.3, color='#ffe66d')
        ax2.plot(list(times), list(lights), color='#ffe66d', linewidth=2)
        ax2.set_ylabel('Light (%)')
        ax2.set_xlabel('Time (min)')
        ax2.set_ylim(0, 100)
    except: pass

ani = FuncAnimation(fig, update, interval=1000, cache_frame_data=False)
plt.tight_layout(rect=[0, 0.05, 1, 0.95])
plt.show()
```

### 프로젝트 2: CO2 농도와 수업 집중도

**연구 질문:** "수업 시간이 지나면서 CO2 농도가 얼마나 올라가고, 환기하면 얼마나 빨리 떨어질까?"

**가설:** "30명 교실에서 환기 없이 40분이 지나면 CO2가 1500ppm을 넘을 것이다."

**센서:** SCD41 CO2 센서 (I2C, 0x62) + OLED

**실험 설계:**

| 항목 | 내용 |
|------|------|
| 독립 변인 | 시간 (수업 시작부터 종료까지) |
| 종속 변인 | CO2 농도 (ppm) |
| 통제 변인 | 학생 수, 교실 크기, 문/창문 상태 |
| 측정 기간 | 수업 1교시 (40분) + 쉬는 시간 (10분) |
| 측정 간격 | 30초 |

:::note[SCD41 사용 시 주의]
SCD41의 `start_periodic_measurement` 명령은 **프로그램 시작 시 한 번만** 호출합니다. 이후 5초마다 자동으로 측정되며, 읽기 명령(`0xEC, 0x05`)만 반복합니다.
:::

#### 전체 코드

```python
from machine import I2C, Pin
import time

# ── 연구 설정 ──
MEASURE_INTERVAL = 30  # 30초마다 측정
LOG_FILE = "co2_research.csv"

# ── I2C 설정 (SCD41 + OLED) ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
scd_addr = 0x62
time.sleep(0.1)
```

### 프로젝트 3: 운동 강도와 가속도 패턴

**연구 질문:** "걷기, 달리기, 제자리 뛰기의 가속도 패턴은 어떻게 다를까?"

**가설:** "달리기의 가속도 진폭은 걷기보다 2배 이상 클 것이다."

**센서:** MPU6050 가속도 센서 (I2C, 0x68) + OLED

**실험 설계:**

| 항목 | 내용 |
|------|------|
| 독립 변인 | 운동 종류 (걷기/달리기/제자리 뛰기) |
| 종속 변인 | 3축 가속도 진폭 (G 단위) |
| 통제 변인 | 센서 부착 위치, 측정 시간 |
| 측정 기간 | 동작별 30초 × 3회 |
| 측정 간격 | 0.2초 (5Hz) |

:::note[MPU6050 센서]
MPU6050은 3축 가속도와 3축 자이로를 동시에 측정할 수 있는 6축 센서입니다. I2C 주소는 0x68이며, 전원 켜면 슬립 모드이므로 0x6B 레지스터에 0x00을 써서 해제해야 합니다.
:::

#### 전체 코드

```python
from machine import I2C, Pin
import time
import math

# ── I2C 설정 (MPU6050 + OLED) ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
mpu_addr = 0x68
time.sleep(0.1)

# MPU6050 초기화
i2c.writeto_mem(mpu_addr, 0x6B, bytes([0x00]))  # 슬립 해제
time.sleep(0.1)

from ssd1306 import SSD1306_I2C
oled = SSD1306_I2C(128, 64, i2c)

def read_accel():
    data = i2c.readfrom_mem(mpu_addr, 0x3B, 6)
    ax = (data[0] << 8 | data[1]) - (65536 if data[0] > 127 else 0)
    ay = (data[2] << 8 | data[3]) - (65536 if data[2] > 127 else 0)
    az = (data[4] << 8 | data[5]) - (65536 if data[4] > 127 else 0)
    # g 단위로 변환
    return ax / 16384, ay / 16384, az / 16384

def classify_motion(magnitude):
    if magnitude < 1.2:
        return "Idle"
    elif magnitude < 2.0:
        return "Walk"
    elif magnitude < 4.0:
        return "Run"
    else:
        return "Jump"

# ── CSV 파일 준비 ──
import os
new_file = "motion_log.csv" not in os.listdir()
f = open("motion_log.csv", "a")
if new_file:
    f.write("time_ms,ax,ay,az,magnitude,motion\n")
start = time.ticks_ms()
mag_hist = []  # 가속도 크기 파형 (최대 60개)

# ── 메인 루프 ──
while True:
    ax, ay, az = read_accel()
    mag = math.sqrt(ax*ax + ay*ay + az*az)
    motion = classify_motion(mag)
    elapsed = time.ticks_diff(time.ticks_ms(), start)

    # 이력 저장
    mag_hist.append(mag)
    if len(mag_hist) > 60:
        mag_hist.pop(0)

    # OLED 대시보드
    oled.fill(0)
    # 헤더 바
    oled.fill_rect(0, 0, 128, 12, 1)
    oled.text("Motion Lab", 24, 2, 0)
    oled.hline(0, 13, 128, 1)
    # 3축 값 컴팩트
    oled.text(f"X:{ax:.1f}", 0, 16)
    oled.text(f"Y:{ay:.1f}", 44, 16)
    oled.text(f"Z:{az:.1f}", 88, 16)
    # 크기 + 분류
    oled.text(f"G:{mag:.2f}", 4, 28)
    oled.text(f"[{motion}]", 72, 28)
    # 실시간 파형 (60포인트, y=40~62)
    oled.hline(0, 39, 128, 1)
    if len(mag_hist) > 1:
        m_max = max(max(mag_hist), 2.0)
        for i in range(len(mag_hist)):
            h = int(min(mag_hist[i] / m_max, 1.0) * 22)
            x = 4 + i * 2
            oled.vline(x, 62 - h, max(h, 1), 1)
    oled.show()

    # CSV 저장
    f.write(f"{elapsed},{ax:.2f},{ay:.2f},{az:.2f},{mag:.2f},{motion}\n")
    f.flush()

    print(f"G:{mag:.2f} [{motion}]")
    time.sleep(0.2)
```

#### 줄별 해설

| 줄 | 코드 | 설명 |
|----|------|------|
| 11 | `writeto_mem(mpu_addr, 0x6B, bytes([0x00]))` | MPU6050의 전원 관리 레지스터에 0을 써서 슬립 모드를 해제합니다. |
| 16-20 | `read_accel()` | 가속도 데이터 6바이트를 읽어 X, Y, Z 축으로 분리합니다. 16384로 나누면 G 단위(1G = 9.8m/s²)가 됩니다. |
| 22-29 | `classify_motion()` | 가속도 크기(magnitude)로 동작을 분류합니다. 정지 시 약 1G(중력), 움직이면 더 커집니다. |
| 37 | `math.sqrt(ax*ax + ay*ay + az*az)` | 3축 가속도의 크기(magnitude)를 구합니다. 이 값은 방향에 관계없이 전체 가속도를 나타냅니다. |

:::

### 자주 만나는 오류 메시지 사전

영어 오류 메시지가 나와도 당황하지 마세요! 아래 표에서 찾아보세요.

| 오류 메시지 | 의미 | 해결 방법 |
|---|---|---|
| `ModuleNotFoundError: No module named 'serial'` | pyserial이 설치되지 않음 | 터미널에서 `pip install pyserial` 실행 |
| `ModuleNotFoundError: No module named 'matplotlib'` | matplotlib이 설치되지 않음 | 터미널에서 `pip install matplotlib` 실행 |
| `OSError: [Errno 2] ENOENT` | 파일을 찾을 수 없음 | CSV 파일 경로 확인 — 스크립트와 같은 폴더에 있는지 확인 |
| `ConnectionRefusedError` | 서버가 실행되지 않음 | `server.py`가 실행 중인지 확인 |
| `OSError: -2` (Pico) | WiFi 네트워크를 찾을 수 없음 | SSID 철자 확인, 2.4GHz WiFi인지 확인 (5GHz는 안 됨) |
| `EHOSTUNREACH` | 서버 PC에 접근 불가 | 같은 WiFi인지 확인, IP 주소가 맞는지 확인 |
| `ImportError: no module named 'ssd1306'` | OLED 라이브러리 미설치 | Thonny → 도구 → 패키지 관리 → `micropython-ssd1306` 설치 |
| `OSError: [Errno 5] EIO` | I2C 통신 오류 | 배선 확인 — SDA/SCL이 바뀌었거나 연결이 헐거울 수 있음 |

## 한 단계 더: WiFi 실시간 통신

지금까지 데이터를 **Pico 내부 CSV 파일**에 저장하고, 나중에 PC로 옮겨 분석했습니다. 하지만 이 방식에는 불편한 점이 있어요.

- USB 케이블을 꽂고 Thonny에서 파일을 다운로드해야 합니다
- 측정하는 동안 실시간으로 PC에서 확인할 수 없습니다
- 전시용 프로젝트에서는 Pico가 멀리 떨어져 있을 수 있습니다

**WiFi 실시간 통신**을 사용하면 Pico가 측정한 데이터를 **무선으로 PC 서버에 전송**하고, 웹 브라우저에서 실시간으로 확인할 수 있습니다.

### 클라이언트-서버 개념

```
┌─────────────┐     WiFi      ┌──────────────┐     브라우저     ┌──────────────┐
│  Pico 2 WH  │ ──────────▶  │  Flask 서버   │  ◀──────────── │ 웹 대시보드   │
│  (클라이언트)  │   POST 전송   │   (PC에서)    │    GET 요청     │  (실시간 확인) │
│  센서 데이터   │              │  데이터 저장   │               │  그래프 표시   │
└─────────────┘              └──────────────┘               └──────────────┘
```

- **클라이언트(Client):** 데이터를 보내는 쪽 → Pico가 센서를 읽고 서버에 전송
- **서버(Server):** 데이터를 받아 저장하고, 요청이 오면 보여주는 쪽 → PC의 Flask 앱
- **대시보드:** 웹 브라우저에서 서버에 접속해 실시간 데이터를 확인

:::note[필요 조건]
Pico 2 WH와 PC가 **같은 WiFi 네트워크**에 연결되어 있어야 합니다. 학교 WiFi나 핫스팟을 사용하세요.
:::

### Step 1: PC에서 Flask 서버 실행

:::note[Flask 서버 설치하기 (처음이라면)]
1. 터미널(명령 프롬프트)을 엽니다
   - Windows: 시작 → "cmd" 검색 → 명령 프롬프트
   - Mac: ⌘+Space → "터미널" 검색
2. 아래 명령어를 입력하고 Enter:
   ```
   pip install flask flask-cors
   ```
   (Mac에서 오류 나면: `pip3 install flask flask-cors`)
3. "Successfully installed..." 메시지가 뜨면 성공!
   - "pip: command not found" 오류? → Python을 먼저 설치하세요 (python.org)
:::

아래 코드를 `server.py`로 저장하고 실행하세요.

```python
# server.py — PC에서 실행하는 데이터 수집 서버
from flask import Flask, request, jsonify
from flask_cors import CORS
from datetime import datetime

app = Flask(__name__)
CORS(app)

# 수신된 데이터를 메모리에 저장 (최근 500개)
sensor_data = []
MAX_DATA = 500

@app.route("/")
def dashboard():
    """웹 대시보드 — 브라우저에서 실시간 확인"""
    return """
    <!DOCTYPE html>
    <html><head><meta charset="utf-8">
    <title>Pico 실시간 대시보드</title>
    <meta http-equiv="refresh" content="5">
    <style>
      body { font-family: sans-serif; max-width: 800px; margin: 40px auto; padding: 0 20px; }
      table { width: 100%; border-collapse: collapse; margin-top: 20px; }
      th, td { border: 1px solid #ddd; padding: 8px; text-align: center; }
      th { background: #4CAF50; color: white; }
      tr:nth-child(even) { background: #f2f2f2; }
      .status { color: #4CAF50; font-weight: bold; }
    </style>
    </head><body>
    <h1>Pico Sensor Dashboard</h1>
    <p class="status">5초마다 자동 새로고침</p>
    <p>수신된 데이터: <strong>""" + str(len(sensor_data)) + """</strong>개</p>
    <table><tr><th>시간</th><th>데이터</th></tr>
    """ + "".join(
        f"<tr><td>{d['time']}</td><td>{d['values']}</td></tr>"
        for d in sensor_data[-20:][::-1]
    ) + """
    </table></body></html>
    """

@app.route("/api/data", methods=["POST"])
def receive_data():
    """Pico에서 보낸 센서 데이터를 수신"""
    data = request.get_json()
    entry = {
        "time": datetime.now().strftime("%H:%M:%S"),
        "values": data
    }
    sensor_data.append(entry)
    if len(sensor_data) > MAX_DATA:
        sensor_data.pop(0)
    print(f"[수신] {entry['time']} → {data}")
    return jsonify({"status": "ok", "count": len(sensor_data)})

@app.route("/api/data", methods=["GET"])
def get_data():
    """저장된 데이터 조회 (최근 N개)"""
    n = request.args.get("n", 50, type=int)
    return jsonify(sensor_data[-n:])

if __name__ == "__main__":
    import socket
    hostname = socket.gethostname()
    local_ip = socket.gethostbyname(hostname)
    print(f"\n{'='*50}")
    print(f"  Flask 서버 시작!")
    print(f"  대시보드:  http://{local_ip}:5000")
    print(f"  API 주소:  http://{local_ip}:5000/api/data")
    print(f"{'='*50}\n")
    app.run(host="0.0.0.0", port=5000, debug=True)
```

2. 위의 서버 코드를 복사하여 붙여넣습니다

   - 저장 위치: 바탕화면에 "pico-server" 폴더를 만들어 저장하면 편해요
4. 서버 실행:
   - 터미널에서: `cd Desktop/pico-server` (Mac: `cd ~/Desktop/pico-server`)

5. 성공하면 이런 메시지가 나타납니다:
   `"🌐 대시보드: http://192.168.x.x:5000"`
   이 주소를 브라우저에 붙여넣으면 대시보드가 보입니다!

:::caution[IP 주소 찾기 — 이것이 핵심!]
Pico가 PC의 서버에 데이터를 보내려면, PC의 IP 주소를 알아야 해요.

**IP 주소란?** 같은 WiFi 안에서 각 기기를 구별하는 "집 주소" 같은 것이에요.

**PC IP 주소 확인 방법:**
- **Windows**: 명령 프롬프트에서 `ipconfig` 입력 → "IPv4 주소" 찾기 (예: 192.168.0.15)
- **Mac**: 시스템 설정 → WiFi → 연결된 네트워크 옆 ⓘ 클릭 → IP 주소 확인

이 IP 주소를 Pico 코드의 `SERVER` 부분에 넣으세요.
⚠️ `localhost`나 `127.0.0.1`은 안 됩니다! 반드시 실제 IP 주소를 사용하세요.
:::

:::caution[Mac 사용자: 포트 5000 충돌]
Mac에서 포트 5000이 이미 사용 중이라는 오류가 나올 수 있어요.
해결: 서버 코드에서 `port=5000`을 `port=5001`로 변경하세요.
Pico 코드의 SERVER 주소에서도 포트를 5001로 맞춰주세요 (예: `http://192.168.0.10:5001`).
:::

### Step 2: Pico에서 WiFi로 데이터 전송

아래 코드를 Pico에 업로드합니다. `WIFI_SSID`, `WIFI_PASS`, `SERVER` 값을 자신의 환경에 맞게 수정하세요.

```python
# main.py — Pico 2 WH에서 실행
import network
import time
import urequests
import json
from machine import I2C, Pin, ADC

# ── WiFi 설정 ──
WIFI_SSID = "학교WiFi이름"
WIFI_PASS = "WiFi비밀번호"
SERVER = "http://192.168.0.10:5000"  # Flask 서버 IP

# ── WiFi 연결 ──
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(WIFI_SSID, WIFI_PASS)

# WiFi 연결 (최대 20초 시도)
print("WiFi 연결 중...")
timeout = 40  # 0.5초 × 40 = 20초
while not wlan.isconnected() and timeout > 0:
    time.sleep(0.5)
    timeout -= 1
    if timeout % 10 == 0:
        print(f"  연결 시도 중... ({(40-timeout)//2}초)")

if not wlan.isconnected():
    print("❌ WiFi 연결 실패! SSID와 비밀번호를 확인하세요")
    print("💡 팁: 핸드폰 핫스팟을 사용해보세요")
    raise SystemExit
else:
    print(f"✅ WiFi 연결 성공! IP: {wlan.ifconfig()[0]}")

# ── 센서 설정 (예: DHT20 온습도) ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
dht_addr = 0x38
light = ADC(Pin(26))
time.sleep(0.1)

def read_dht20():
    i2c.writeto(dht_addr, bytes([0xAC, 0x33, 0x00]))
    time.sleep(0.08)
    data = i2c.readfrom(dht_addr, 7)
    humi = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4)) / 1048576 * 100
    temp = (((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]) / 1048576 * 200 - 50
    return round(temp, 1), round(humi, 1)

# ── 서버로 데이터 전송 ──
def send_to_server(data_dict):
    try:
        res = urequests.post(
            f"{SERVER}/api/data",
            json=data_dict,
            headers={"Content-Type": "application/json"}
        )
        print(f"  서버 응답: {res.status_code}")
        res.close()
        return True
    except Exception as e:
        print(f"  전송 실패: {e}")
        return False

# ── 메인 루프 ──
count = 0
while True:
    try:
        temp, humi = read_dht20()
        light_pct = round(light.read_u16() / 65535 * 100, 1)
        count += 1

        data = {
            "count": count,
            "temp": temp,
            "humi": humi,
            "light": light_pct
        }

        print(f"#{count} | T:{temp}C H:{humi}% L:{light_pct}%")
        send_to_server(data)

    except Exception as e:
        print(f"오류: {e}")

    time.sleep(10)  # 10초마다 전송
```

### Step 3: 실시간 확인

1. PC에서 `python server.py` 실행
2. Pico에 WiFi 코드 업로드 후 실행
3. 웹 브라우저에서 `http://서버IP:5000` 접속
4. 데이터가 5초마다 자동으로 갱신됩니다

:::tip[WiFi 전송 + CSV 저장을 동시에]
안정성을 위해 WiFi 전송과 Pico 내부 CSV 저장을 **동시에** 하는 것이 좋습니다. WiFi가 끊겨도 CSV에는 데이터가 남아 있으니까요. 위 코드에 앞서 배운 CSV 저장 코드를 함께 넣어보세요.
:::

:::caution[주의사항]
- `SERVER` 변수에는 PC의 **로컬 IP 주소**를 입력하세요 (`localhost`나 `127.0.0.1`이 아닙니다).
- 학교 WiFi가 기기 간 통신을 차단하는 경우, 핸드폰 핫스팟을 사용해보세요.
- `urequests` 모듈이 없다면 Thonny에서 "도구 > 패키지 관리"로 설치하세요.
:::
