### 프로젝트 1: 학교 공기질 종합 모니터링 시스템

**문제:** 교실마다 공기질이 다른데, 언제 어디서 환기해야 할지 데이터가 없다.

**센서 구성:**
- SCD41 CO2 센서 (I2C, 0x62) — CO2 농도
- DHT20 온습도 센서 (I2C, 0x38) — 온도, 습도
- 먼지 센서 (ADC, GP28) — 간이 미세먼지 측정
- OLED (I2C, 0x3C) — 대시보드 표시

**ML 활용:**
- 센서 데이터를 10분 단위로 수집하여 CSV로 저장
- "Good / Fair / Poor / Bad" 4개 등급으로 공기질 인덱스(AQI) 점수 산출
- Teachable Machine 또는 규칙 기반 분류 모델 적용

**이중 대시보드:**
- OLED: CO2 + 먼지 + 온도 + AQI 점수 + 막대그래프
- PC: Serial 출력을 matplotlib으로 실시간 그래프

:::note[핀 배치]
- DHT20, SCD41, OLED: I2C1 (SDA=GP6, SCL=GP7) — 주소가 모두 다르므로 공존 가능
- 먼지 센서: GP28 (ADC2) — A0(GP26)과 A1(GP27)은 다른 센서용으로 예약
:::

#### 전체 코드

```python
from machine import I2C, Pin, ADC
import time

# ── 센서 설정 ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
dht_addr = 0x38
scd_addr = 0x62
dust = ADC(Pin(28))  # 미세먼지 센서 (ADC2 — GP28)
time.sleep(0.1)

from ssd1306 import SSD1306_I2C
oled = SSD1306_I2C(128, 64, i2c)

def read_dht20():
    i2c.writeto(dht_addr, bytes([0xAC, 0x33, 0x00]))
    time.sleep(0.08)
    data = i2c.readfrom(dht_addr, 7)
    humi = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4)) / 1048576 * 100
    temp = (((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]) / 1048576 * 200 - 50
    return round(temp, 1), round(humi, 1)

def read_scd41():
    i2c.writeto(scd_addr, bytes([0xEC, 0x05]))
    time.sleep(0.01)
    data = i2c.readfrom(scd_addr, 9)
    return (data[0] << 8) | data[1]

# SCD41 주기 측정 시작 (한 번만 호출)
i2c.writeto(scd_addr, bytes([0x21, 0xB1]))
print("SCD41 준비 중... 5초 대기")
time.sleep(5)

def read_dust():
    raw = dust.read_u16()
    voltage = raw / 65535 * 3.3
    dust_ug = max(0, (voltage - 0.6) * 170)  # GP2Y 센서 근사식
    return round(dust_ug, 0)

def air_quality_index(co2, dust_ug, temp, humi):
    score = 100
    if co2 > 2000: score -= 50
    elif co2 > 1500: score -= 40
    elif co2 > 1000: score -= 30
    elif co2 > 800: score -= 15
    if dust_ug > 150: score -= 40
    elif dust_ug > 75: score -= 30
    elif dust_ug > 35: score -= 15
    if temp < 18 or temp > 28: score -= 10
    if humi < 30 or humi > 70: score -= 10
    return max(0, score)

def aqi_grade(score):
    if score >= 80: return "Good"
    elif score >= 60: return "Fair"
    elif score >= 40: return "Poor"
    else: return "Bad!"

# ── CSV 파일 준비 ──
f = open("air_quality.csv", "a")
f.write("분,CO2,온도,습도,미세먼지,AQI점수,등급\n")
count = 0
start = time.ticks_ms()
aqi_hist = []  # AQI 트렌드 (최대 40개)

# ── 메인 루프 ──
while True:
    temp, humi = read_dht20()
    co2 = read_scd41()
    dust_ug = read_dust()
    aqi = air_quality_index(co2, dust_ug, temp, humi)
    grade = aqi_grade(aqi)
    count += 1
    elapsed_min = time.ticks_diff(time.ticks_ms(), start) // 60000

    # AQI 이력
    aqi_hist.append(aqi)
    if len(aqi_hist) > 40:
        aqi_hist.pop(0)

    # OLED 이중 대시보드
    oled.fill(0)
    # 외곽 프레임
    oled.rect(0, 0, 128, 64, 1)
    # 헤더 바
    oled.fill_rect(1, 1, 126, 11, 1)
    oled.text("Air Quality", 24, 2, 0)
    # 왼쪽 패널: CO2 + Dust
    oled.text(f"CO2:{co2}", 4, 14)
    co2_w = min(int(co2 / 2000 * 40), 40)
    oled.rect(4, 24, 42, 6, 1)
    oled.fill_rect(5, 25, co2_w, 4, 1)
    oled.text(f"D:{dust_ug:.0f}", 4, 32)
    d_w = min(int(dust_ug / 200 * 40), 40)
    oled.rect(4, 42, 42, 6, 1)
    oled.fill_rect(5, 43, d_w, 4, 1)
    # 오른쪽 패널: Temp + Humi
    oled.vline(64, 13, 38, 1)
    oled.text(f"T:{temp}C", 68, 14)
    oled.text(f"H:{humi}%", 68, 26)
    # AQI 스코어 바 (하단)
    oled.hline(1, 50, 126, 1)
    oled.text(f"AQI:{aqi}", 4, 53)
    oled.text(f"[{grade}]", 76, 53)
    aqi_w = int(aqi / 100 * 120)
    oled.fill_rect(4, 62, aqi_w, 1, 1)
    oled.show()

    # CSV + Serial (PC 이중 대시보드)
    f.write(f"{elapsed_min},{co2},{temp},{humi},{dust_ug:.0f},{aqi},{grade}\n")
    f.flush()
    # Serial 출력 (PC 실시간 그래프용)
    print(f"CO2:{co2},T:{temp},H:{humi},D:{dust_ug:.0f},AQI:{aqi}")
    time.sleep(10)
```

```python
# PC 실시간 대시보드 — 학교 공기질 모니터링
# 실행: python air_quality_pc.py

import serial
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
from collections import deque
import time

PORT = '/dev/tty.usbmodem1101'  # 환경에 맞게 수정
BAUD = 115200
ser = serial.Serial(PORT, BAUD, timeout=1)
time.sleep(2)

MAX_POINTS = 200
idx = deque(maxlen=MAX_POINTS)
co2s = deque(maxlen=MAX_POINTS)
dusts = deque(maxlen=MAX_POINTS)
temps_d = deque(maxlen=MAX_POINTS)
humis_d = deque(maxlen=MAX_POINTS)
aqis = deque(maxlen=MAX_POINTS)
count = 0

plt.style.use('dark_background')
fig, ((ax_co2, ax_dust), (ax_th, ax_aqi)) = plt.subplots(2, 2, figsize=(14, 8))
fig.suptitle('Air Quality Monitoring Station', fontsize=14, color='cyan')
ax_th_r = ax_th.twinx()

def update(frame):
    global count
    try:
        line = ser.readline().decode('utf-8').strip()
        if not line or not line.startswith('CO2:'): return
        # 파싱: CO2:800,T:24.5,H:55,D:25,AQI:85
        parts = dict(p.split(':') for p in line.split(','))
        co2 = int(parts['CO2'])
        temp = float(parts['T'])
        humi = float(parts['H'])
        dust = float(parts['D'])
        aqi = int(parts['AQI'])
        count += 1
        idx.append(count)
        co2s.append(co2)
        dusts.append(dust)
        temps_d.append(temp)
        humis_d.append(humi)
        aqis.append(aqi)

        x = list(idx)

        # CO2 차트
        ax_co2.clear()
        ax_co2.set_title('CO2 (ppm)', color='#ff6b6b')
        ax_co2.axhspan(0, 600, alpha=0.1, color='green')
        ax_co2.axhspan(600, 1000, alpha=0.1, color='yellow')
        ax_co2.axhspan(1000, 1500, alpha=0.1, color='orange')
        ax_co2.axhspan(1500, 3000, alpha=0.1, color='red')
        ax_co2.plot(x, list(co2s), color='#ff6b6b', linewidth=2)
        ax_co2.set_ylim(300, max(1600, max(co2s) + 100) if co2s else 1600)
        ax_co2.grid(True, alpha=0.3)

        # Dust 차트
        ax_dust.clear()
        ax_dust.set_title('Dust (ug/m3)', color='#ffaa44')
        ax_dust.axhspan(0, 35, alpha=0.1, color='green')
        ax_dust.axhspan(35, 75, alpha=0.1, color='yellow')
        ax_dust.axhspan(75, 500, alpha=0.1, color='red')
        ax_dust.plot(x, list(dusts), color='#ffaa44', linewidth=2)
        ax_dust.grid(True, alpha=0.3)

        # Temp/Humi 차트
        ax_th.clear()
        ax_th_r.clear()
        ax_th.set_title('Temp & Humidity', color='#4ecdc4')
        ax_th.plot(x, list(temps_d), color='#ff6b6b', linewidth=1.5, label='Temp(C)')
        ax_th_r.plot(x, list(humis_d), color='#4ecdc4', linewidth=1.5, label='Humi(%)')
        ax_th.set_ylabel('Temp', color='#ff6b6b')
        ax_th_r.set_ylabel('Humi', color='#4ecdc4')
        ax_th.legend(loc='upper left', fontsize=8)
        ax_th_r.legend(loc='upper right', fontsize=8)
        ax_th.grid(True, alpha=0.3)

        # AQI 트렌드
        ax_aqi.clear()
        ax_aqi.set_title('AQI Score', color='cyan')
        aqi_list = list(aqis)
        ax_aqi.axhspan(80, 100, alpha=0.1, color='green')
        ax_aqi.axhspan(60, 80, alpha=0.1, color='yellow')
        ax_aqi.axhspan(40, 60, alpha=0.1, color='orange')
        ax_aqi.axhspan(0, 40, alpha=0.1, color='red')
        ax_aqi.plot(x, aqi_list, color='cyan', linewidth=2)
        ax_aqi.fill_between(x, aqi_list, alpha=0.15, color='cyan')
        ax_aqi.set_ylim(0, 100)
        ax_aqi.grid(True, alpha=0.3)
        if aqi_list:
            ax_aqi.text(0.98, 0.95, f'{aqi_list[-1]}', transform=ax_aqi.transAxes,
                        ha='right', va='top', fontsize=20, color='cyan')
    except: pass

ani = FuncAnimation(fig, update, interval=1000, cache_frame_data=False)
plt.tight_layout()
plt.show()
```

### 프로젝트 2: 교실 에너지 절약 자동화 시스템

**문제:** 교실 조명과 환기가 필요 없을 때도 켜져 있어 에너지가 낭비된다.

**센서 구성:**
- 빛 센서 (ADC, GP26) — 자연광 측정
- PIR 인체감지 센서 (Digital, GP16) — 사람 존재 감지
- DHT20 온습도 센서 (I2C, 0x38) — 냉난방 효율 측정
- 릴레이 (Digital, GP18) — 조명/환풍기 자동 제어
- OLED (I2C, 0x3C) — 대시보드 표시

**ML 활용:**
- "교실 사용 중" / "비어 있음" / "쉬는 시간" 3개 클래스
- PIR 감지 + 빛 패턴 + 시간 정보로 분류
- 비어 있으면 → "조명 OFF 권장" 알림

:::caution[핀 충돌 해결]
PIR 센서가 GP16을 사용합니다. 릴레이의 기본 핀도 GP16이므로, 이 프로젝트에서는 릴레이를 **GP18**에 연결합니다.
:::

#### 전체 코드

```python
from machine import I2C, Pin, ADC
import time

# ── 센서 설정 ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
dht_addr = 0x38
light = ADC(Pin(26))  # Grove A0 포트
pir = Pin(16, Pin.IN)   # PIR 인체감지 (Grove D16)
relay = Pin(18, Pin.OUT) # 릴레이 (GP18 — PIR이 GP16 사용하므로 대체 핀)
time.sleep(0.1)

from ssd1306 import SSD1306_I2C
oled = SSD1306_I2C(128, 64, i2c)

def read_dht20():
    i2c.writeto(dht_addr, bytes([0xAC, 0x33, 0x00]))
    time.sleep(0.08)
    data = i2c.readfrom(dht_addr, 7)
    humi = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4)) / 1048576 * 100
    temp = (((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]) / 1048576 * 200 - 50
    return round(temp, 1), round(humi, 1)

# 자동 제어 로직
EMPTY_TIMEOUT = 300  # 5분 동안 사람 없으면 절전
last_motion = time.ticks_ms()
relay_on = False

# ── CSV 파일 준비 ──
f = open("energy_log.csv", "a")
f.write("분,온도,습도,조도,재실,릴레이,절약시간(초)\n")
save_seconds = 0
start = time.ticks_ms()

# ── 메인 루프 ──
while True:
    temp, humi = read_dht20()
    light_pct = round(light.read_u16() / 65535 * 100, 1)
    occupied = pir.value() == 1
    now = time.ticks_ms()

    if occupied:
        last_motion = now

    idle_sec = time.ticks_diff(now, last_motion) // 1000

    # 자동 제어
    if not occupied and idle_sec > EMPTY_TIMEOUT:
        relay.value(0)
        relay_on = False
        save_seconds += 2
    elif light_pct < 30 and occupied:
        relay.value(1)
        relay_on = True
    elif light_pct > 70:
        relay.value(0)
        relay_on = False
        save_seconds += 2

    elapsed_min = time.ticks_diff(now, start) // 60000

    # OLED 대시보드
    oled.fill(0)
    # 외곽 프레임
    oled.rect(0, 0, 128, 64, 1)
    # 헤더 바
    oled.fill_rect(1, 1, 126, 11, 1)
    oled.text("EnergySaver", 20, 2, 0)
    # 상태 아이콘 행
    oled.text(f"Person:{'Y' if occupied else 'N'}", 4, 14)
    oled.text(f"Relay:{'ON' if relay_on else 'OFF'}", 4, 24)
    # 조도 레벨 바
    oled.text(f"Light:{light_pct:.0f}%", 4, 36)
    l_w = int(min(light_pct, 100) / 100 * 56)
    oled.rect(68, 36, 58, 8, 1)
    oled.fill_rect(69, 37, l_w, 6, 1)
    # 절약 카운터 크게
    oled.hline(1, 48, 126, 1)
    oled.text(f"Saved:{save_seconds//60}min", 4, 52)
    oled.show()

    # CSV 저장
    f.write(f"{elapsed_min},{temp},{humi},{light_pct},{'Y' if occupied else 'N'},{'ON' if relay_on else 'OFF'},{save_seconds}\n")
    f.flush()

    print(f"T:{temp} L:{light_pct:.0f}% P:{'Y' if occupied else 'N'} R:{'ON' if relay_on else 'OFF'}")
    time.sleep(2)
```

#### 자동 제어 로직 해설

```
1. 사람이 없고, 5분 이상 지남 → 릴레이 OFF (절전)
2. 사람이 있고, 어두움(30% 미만) → 릴레이 ON (조명 켜기)
3. 밝음(70% 이상) → 릴레이 OFF (자연광으로 충분)
```

### 프로젝트 3: 센서 기반 동작 분류 ML 시스템

**문제:** 착용형 센서로 사람의 동작(앉기/걷기/달리기)을 자동 분류할 수 있을까?

**센서 구성:**
- MPU6050 가속도 센서 (I2C, 0x68) — 3축 가속도
- OLED (I2C, 0x3C) — 대시보드 표시

**ML 활용:**
- 동작별 가속도 데이터를 20Hz로 수집 (각 동작 100개 샘플)
- "Idle / Walk / Shake / Jump" 4개 클래스
- Teachable Machine에 CSV 업로드 또는 규칙 기반 분류

:::note[ML 파이프라인]
```
1. Pico에서 가속도 데이터 CSV 수집 (동작별)
2. CSV를 PC로 전송
3. Teachable Machine에 업로드하여 학습
4. 학습된 규칙을 Pico 코드에 반영
5. 실시간 분류 결과를 대시보드에 표시
```
:::

#### 전체 코드

이 코드는 **두 가지 모드**로 동작합니다:
- `MODE = "collect"`: ML 학습용 데이터 수집
- `MODE = "predict"`: 규칙 기반 실시간 동작 분류

```python
from machine import I2C, Pin
import time
import math

# ── I2C 설정 ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
mpu_addr = 0x68
time.sleep(0.1)

# MPU6050 초기화
i2c.writeto_mem(mpu_addr, 0x6B, bytes([0x00]))
time.sleep(0.1)

from ssd1306 import SSD1306_I2C
oled = SSD1306_I2C(128, 64, i2c)

def read_accel():
    data = i2c.readfrom_mem(mpu_addr, 0x3B, 6)
    ax = (data[0] << 8 | data[1]) - (65536 if data[0] > 127 else 0)
    ay = (data[2] << 8 | data[3]) - (65536 if data[2] > 127 else 0)
    az = (data[4] << 8 | data[5]) - (65536 if data[4] > 127 else 0)
    return ax / 16384, ay / 16384, az / 16384

# ── 데이터 수집 모드 ──
# Teachable Machine 학습용 데이터 수집
MODE = "collect"  # "collect" 또는 "predict"
LABEL = "idle"    # 수집할 동작 이름
SAMPLES = 100     # 수집할 샘플 수

def collect_data(label, n_samples):
    print(f"[{label}] 데이터 수집 시작! 3초 후...")
    time.sleep(3)
    f = open(f"ml_{label}.csv", "a")
    f.write("ax,ay,az,magnitude\n")

    for i in range(n_samples):
        ax, ay, az = read_accel()
        mag = math.sqrt(ax*ax + ay*ay + az*az)
        f.write(f"{ax:.3f},{ay:.3f},{az:.3f},{mag:.3f}\n")
        pct = int((i + 1) / n_samples * 100)

        # OLED 수집 대시보드
        oled.fill(0)
        oled.fill_rect(0, 0, 128, 12, 1)
        oled.text("Collect", 36, 2, 0)
        oled.hline(0, 13, 128, 1)
        oled.text(f"Label: {label}", 4, 16)
        oled.text(f"{i+1}/{n_samples}", 4, 28)
        oled.text(f"{pct}%", 100, 28)
        # 진행 막대 (프레임 + 채움)
        oled.rect(4, 42, 120, 10, 1)
        oled.fill_rect(5, 43, int(pct / 100 * 118), 8, 1)
        oled.text(f"G:{mag:.2f}", 4, 56)
        oled.show()

        time.sleep(0.05)  # 20Hz 샘플링

    f.close()
    print(f"[{label}] {n_samples}개 수집 완료!")

def predict_mode():
    mag_hist = []  # 크기 파형 (최대 60개)
    # 간단한 규칙 기반 분류 (ML 모델 적용 전 테스트용)
    while True:
        ax, ay, az = read_accel()
        mag = math.sqrt(ax*ax + ay*ay + az*az)

        if mag < 1.1:
            label = "Idle"
        elif mag < 2.0:
            label = "Walk"
        elif mag < 4.0:
            label = "Shake"
        else:
            label = "Jump"

        mag_hist.append(mag)
        if len(mag_hist) > 60:
            mag_hist.pop(0)

        # OLED 예측 대시보드
        oled.fill(0)
        # 헤더 바
        oled.fill_rect(0, 0, 128, 12, 1)
        oled.text("ML Predict", 24, 2, 0)
        oled.hline(0, 13, 128, 1)
        # 크기 게이지 (수평 바)
        oled.text(f"G:{mag:.2f}", 4, 16)
        gauge_w = min(int(mag / 5.0 * 54), 54)
        oled.rect(68, 16, 56, 8, 1)
        # 게이지 표시선
        oled.vline(68 + int(1.1/5*56), 16, 8, 1)
        oled.vline(68 + int(2.0/5*56), 16, 8, 1)
        oled.vline(68 + int(4.0/5*56), 16, 8, 1)
        oled.fill_rect(69, 17, gauge_w, 6, 1)
        # 분류 결과 크게
        oled.text(f"=> {label}", 20, 28)
        # 실시간 파형 (하단)
        oled.hline(0, 39, 128, 1)
        if len(mag_hist) > 1:
            m_max = max(max(mag_hist), 2.0)
            for i in range(len(mag_hist)):
                h = int(min(mag_hist[i] / m_max, 1.0) * 22)
                x = 4 + i * 2
                oled.vline(x, 62 - h, max(h, 1), 1)
        oled.show()

        # Serial로 PC 전송 (ML 모델 연동용)
        print(f"{ax:.3f},{ay:.3f},{az:.3f},{mag:.3f},{label}")
        time.sleep(0.1)

# ── 실행 ──
if MODE == "collect":
    collect_data(LABEL, SAMPLES)
else:
    predict_mode()
```

## ☁️ 최종 목표: Supabase 클라우드 대시보드

지금까지 여러분의 시스템은 **Pico → Serial → PC**로 연결되어 있었어요. 훌륭하지만, 진정한 IoT 시스템은 한 단계 더 나아갑니다.

> **클라우드에 데이터를 저장하면, 어디서든 어떤 기기에서든 실시간으로 확인할 수 있어요.**

학교 3층 과학실에 설치한 Pico가 측정한 공기질을 — 집에서 스마트폰으로 확인할 수 있다면? 이것이 진짜 IoT입니다.

```
[기존 아키텍처]
 Pico → Serial → PC (1:1, 같은 교실에서만)

[클라우드 아키텍처]
 Pico → WiFi → Supabase ← 웹 대시보드 (어디서든 접속)
      → WiFi → Supabase ← 스마트폰 (실시간 알림)
```

### 왜 Supabase인가요?

| | 로컬 CSV | Supabase 클라우드 |
|---|---|---|
| **접근성** | USB 연결된 PC에서만 | 인터넷 되는 곳이면 어디서든 |
| **실시간성** | Serial 직접 연결 필요 | 웹소켓으로 자동 업데이트 |
| **저장 용량** | Pico 플래시 2MB 제한 | 500MB (무료 플랜) |
| **다중 기기** | 불가 | Pico 여러 대 → 하나의 대시보드 |
| **비용** | 무료 | 무료 (월 50만 API 호출) |

Supabase는 오픈소스 Firebase 대안으로, PostgreSQL 데이터베이스와 실시간 구독 기능을 무료로 제공합니다.

### Step 2: 센서 데이터 테이블 생성

프로젝트가 생성되면 **SQL Editor**에서 아래 SQL을 실행하세요.

:::note[SQL Editor 사용하기]
**SQL이란?** 데이터베이스에 명령을 내리는 언어예요. 어렵지 않아요, 복사+붙여넣기만 하면 됩니다!

1. 왼쪽 메뉴에서 **"SQL Editor"** 아이콘 클릭 (데이터베이스 아이콘 모양)
2. "New query" 버튼 클릭
3. 아래 SQL 코드를 **전체 복사**하여 붙여넣기
4. **"Run"** 버튼 클릭 (또는 Ctrl+Enter / Cmd+Enter)
5. 아래쪽에 "Success. No rows returned" 메시지가 나오면 성공!
   - 이 메시지가 정상이에요 — 테이블을 "만드는" 명령이라 돌려줄 데이터가 없는 거예요
   - 만약 "ERROR" 메시지가 나오면: "relation already exists"는 이미 만들어져 있다는 뜻이니 괜찮아요
:::

```sql
-- 센서 데이터 저장 테이블
CREATE TABLE sensor_data (
  id BIGSERIAL PRIMARY KEY,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  device_id TEXT DEFAULT 'pico-01',
  temp REAL,
  humi REAL,
  co2 INTEGER,
  light REAL,
  sound REAL,
  dust REAL,
  extra JSONB
);

-- 실시간 구독을 위한 설정
ALTER TABLE sensor_data REPLICA IDENTITY FULL;
```

:::tip[테이블 구조 설명]
- `device_id`: Pico 기기별 식별자 — 여러 대를 구분할 수 있어요
- `extra`: JSONB 타입으로, 프로젝트별 추가 데이터를 자유롭게 저장
- `created_at`: 자동으로 현재 시각이 기록되어 시계열 분석에 활용
:::

:::tip[REPLICA IDENTITY FULL이란?]
`ALTER TABLE sensor_data REPLICA IDENTITY FULL;` — 이 명령은 실시간 업데이트 기능을 켜는 설정이에요.
이 줄이 없으면 대시보드가 새 데이터를 자동으로 감지하지 못합니다.
:::

### Step 3: API 키 확인 및 RLS 보안 설정

:::caution[API 키 찾기 — 정확히 따라하세요!]
1. 왼쪽 메뉴 맨 아래 **"Project Settings"** (톱니바퀴 아이콘) 클릭
2. 왼쪽 하위 메뉴에서 **"API"** 클릭
3. 두 가지를 복사합니다:

**Project URL** (프로젝트 주소):
- "Project URL" 항목의 `https://xxxxx.supabase.co` 옆 복사 버튼 클릭
- 이것을 Pico 코드의 `SUPABASE_URL`에 붙여넣기

**anon public key** (공개 키):
- "Project API keys" 섹션에서 **"anon" "public"** 이라고 적힌 키 옆 복사 버튼 클릭
- 이 키는 매우 길어요 (200자 이상). 전체를 복사해야 합니다!
- 이것을 Pico 코드의 `SUPABASE_KEY`에 붙여넣기

**주의**: "service_role" "secret" 키는 절대 복사하지 마세요! 이것은 관리자 전용 키입니다.
:::

그 다음, 데이터를 안전하게 보호하기 위해 **RLS(Row Level Security)** 정책을 설정합니다.

:::note[보안 설정 (RLS)]
**RLS란?** "Row Level Security"의 약자로, 데이터베이스의 보안 규칙이에요.
쉽게 말하면: "누가 데이터를 볼 수 있고, 누가 추가할 수 있는지" 정하는 규칙입니다.

우리가 설정할 규칙:
- 누구나 데이터를 **볼 수** 있음 (대시보드에서 보려고)
- 누구나 데이터를 **추가할 수** 있음 (Pico가 보내려고)
- 삭제나 수정은 안 됨 (데이터 보호)

이 SQL도 SQL Editor에서 새 쿼리로 실행하세요 (위의 CREATE TABLE과 같은 곳에 이어서 붙이지 마세요!)
:::

SQL Editor에서 실행하세요:

```sql
-- RLS 활성화
ALTER TABLE sensor_data ENABLE ROW LEVEL SECURITY;

-- 누구나 데이터를 읽을 수 있게 (대시보드 공개용)
CREATE POLICY "Allow public read"
  ON sensor_data FOR SELECT
  USING (true);

-- Pico에서 데이터를 삽입할 수 있게
CREATE POLICY "Allow insert from anon"
  ON sensor_data FOR INSERT
  WITH CHECK (true);
```

:::caution[보안 참고]
`anon key`는 공개 키입니다. 브라우저 코드에 포함해도 괜찮지만, 반드시 RLS 정책으로 어떤 작업이 허용되는지 제한해야 합니다. 위 설정은 **읽기와 삽입만 허용**하므로, 데이터 삭제나 수정은 불가능합니다.

프로덕션 환경에서는 API Proxy 서버를 두어 서버 측에서 `service_role` 키를 사용하는 것이 더 안전합니다. 하지만 학교 프로젝트에서는 위 설정으로 충분합니다.
:::

```python
    return True

# ── Supabase에 데이터 전송 ──
def send_to_supabase(temp, humi, co2=None, light=None, dust=None):
    url = f"{SUPABASE_URL}/rest/v1/sensor_data"
    headers = {
        "apikey": SUPABASE_KEY,
        "Authorization": f"Bearer {SUPABASE_KEY}",
        "Content-Type": "application/json",
        "Prefer": "return=minimal"
    }
    payload = {
        "device_id": DEVICE_ID,
        "temp": temp,
        "humi": humi
    }
    # 사용하는 센서에 따라 추가
    if co2 is not None:
        payload["co2"] = co2
    if light is not None:
        payload["light"] = light
    if dust is not None:
        payload["dust"] = dust

    try:
        res = urequests.post(url, json=payload, headers=headers)
        print(f"Supabase 전송: {res.status_code}")
        res.close()
        return res.status_code == 201
    except Exception as e:
        print(f"전송 실패: {e}")
        return False

# ── 센서 설정 (프로젝트에 맞게 수정) ──
i2c = I2C(1, sda=Pin(6), scl=Pin(7), freq=100000)
dht_addr = 0x38

def read_dht20():
    i2c.writeto(dht_addr, bytes([0xAC, 0x33, 0x00]))
    time.sleep(0.08)
    data = i2c.readfrom(dht_addr, 7)
    humi = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4)) / 1048576 * 100
    temp = (((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]) / 1048576 * 200 - 50
    return round(temp, 1), round(humi, 1)

# ── OLED (선택) ──
from ssd1306 import SSD1306_I2C
oled = SSD1306_I2C(128, 64, i2c)

# ── 메인 루프 ──
if connect_wifi():
    send_count = 0
    fail_count = 0

    while True:
        try:
            temp, humi = read_dht20()
            send_count += 1

            # Supabase로 전송
            success = send_to_supabase(temp, humi)
            if not success:
                fail_count += 1

            # OLED 표시
            oled.fill(0)
            oled.text("Cloud Monitor", 8, 0)
            oled.text(f"T:{temp}C H:{humi}%", 4, 16)
            oled.text(f"Send: #{send_count}", 4, 32)
            if success:
                oled.text("OK", 100, 32)
            else:
                oled.text("FAIL", 88, 32)
            oled.text(f"Fail: {fail_count}", 4, 48)
            oled.show()

            print(f"#{send_count} T:{temp} H:{humi} -> {'OK' if success else 'FAIL'}")

        except Exception as e:
            print(f"Error: {e}")
            fail_count += 1

        time.sleep(30)  # 30초마다 전송 (무료 플랜 API 제한 고려)
else:
    # WiFi 실패 시 오프라인 모드
    oled.fill(0)
    oled.text("WiFi Failed", 16, 20)
    oled.text("Offline Mode", 12, 36)
    oled.show()
```

:::note[코드 커스터마이징]
위 코드는 DHT20 온습도 센서만 사용하는 기본형입니다. 본인의 프로젝트에 맞게 센서를 추가하세요:
- 공기질 모니터링 → `co2`, `dust` 필드 추가
- 에너지 절약 → `light` 필드 추가, `extra`에 릴레이 상태 저장
- 동작 분류 → `extra`에 가속도 데이터와 분류 결과 저장
:::

### 전시 대시보드 — 학교 IoT 네트워크

이 시스템의 가장 멋진 활용은 **학교 전시**입니다.

```
[과학실]                [운동장]              [교무실]
 Pico-01               Pico-02              Pico-03
 온습도+CO2             온도+빛              온습도+소음
   │                     │                     │
   └───── WiFi ──────────┴──────── WiFi ───────┘
                         │
                    ☁️ Supabase
                         │
              ┌──────────┼──────────┐
           [교실 TV]  [스마트폰]  [PC 대시보드]
           실시간 표시  알림 확인   데이터 분석
```

각 Pico에 다른 `DEVICE_ID`를 설정하면, 하나의 Supabase 테이블에 모든 데이터가 모이고, 대시보드에서 기기별로 필터링할 수 있습니다.

:::tip[전시 아이디어]
1. **학교 환경 지도**: 각 교실의 공기질을 실시간으로 보여주는 학교 평면도 대시보드
2. **환경 알림 시스템**: CO2가 1000ppm을 넘으면 "환기 필요!" 알림
3. **일일 보고서**: 하루 동안 수집된 데이터를 자동으로 정리하여 보고서 생성
4. **QR 코드 접속**: 대시보드 URL을 QR 코드로 만들어 복도에 부착 — 누구나 스마트폰으로 확인
:::

### 자주 만나는 오류와 해결법

| 오류 상황 | 원인 | 해결 방법 |
|---|---|---|
| Pico에서 "WiFi 연결 실패" | SSID/비밀번호 오류 | 철자 확인, 2.4GHz WiFi인지 확인 |
| Pico에서 "전송 실패" 반복 | Supabase URL/KEY 오류 | Settings → API에서 다시 복사 |
| 대시보드에 데이터가 안 나옴 | URL/KEY가 HTML에도 필요 | dashboard.html의 URL과 KEY도 교체했는지 확인 |
| SQL "permission denied" | RLS 설정 안 됨 | RLS SQL 코드를 실행했는지 확인 |
| SQL "relation already exists" | 이미 테이블이 있음 | 정상! 무시해도 됩니다 |
| `ModuleNotFoundError: serial` | pyserial 미설치 | `pip install pyserial` (pyserial입니다, serial 아님!) |
| Thonny와 PC 스크립트 충돌 | 시리얼 포트 점유 | Thonny를 완전히 닫은 후 PC 스크립트 실행 |
| 대시보드 실시간 갱신 안 됨 | REPLICA IDENTITY 미설정 | `ALTER TABLE sensor_data REPLICA IDENTITY FULL;` 실행 |
