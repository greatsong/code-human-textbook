**[코드: step1_led_basic.py]**
```python
# === RGB LED Stick 기본 제어 ===
# NeoPixel LED의 색상을 바꿔보는 코드

from machine import Pin
from neopixel import NeoPixel
import time

# --- LED 설정 ---
# D16 포트 = GP16, LED 10개
NUM_LEDS = 10
np = NeoPixel(Pin(16), NUM_LEDS)

# --- 기본 제어 ---

# 1) 전체 초록색 켜기
for i in range(NUM_LEDS):
    np[i] = (0, 255, 0)  # (R, G, B)
np.write()  # 반드시 write() 해야 적용!
time.sleep(2)

# 2) 전체 노란색
for i in range(NUM_LEDS):
    np[i] = (255, 255, 0)
np.write()
time.sleep(2)

# 3) 전체 빨간색
for i in range(NUM_LEDS):
    np[i] = (255, 0, 0)
np.write()
time.sleep(2)

# 4) 전체 끄기
for i in range(NUM_LEDS):
    np[i] = (0, 0, 0)
np.write()

print("LED 테스트 완료!")
```

```python
# === 전자담배 검출 시스템 v1 ===
# MQ2 센서 값에 따라 LED 색상이 바뀌는 검출기

from machine import ADC, Pin
from neopixel import NeoPixel
import time

# --- 하드웨어 설정 ---
gas_sensor = ADC(Pin(26))      # A0: MQ2 가스 센서
NUM_LEDS = 10
np = NeoPixel(Pin(16), NUM_LEDS)  # D16: RGB LED Stick

# --- 임계값 설정 ---
# ※ 환경에 따라 조정 필요! 먼저 step1으로 평상시 값을 확인하세요
THRESHOLD_WARN = 20000    # 이 이상이면 주의 (노랑)
THRESHOLD_ALERT = 35000   # 이 이상이면 경보 (빨강)

# --- 이동 평균 함수 (1차시 복습) ---
def read_average(sensor, count=10):
    total = 0
    for i in range(count):
        total += sensor.read_u16()
        time.sleep_ms(10)
    return total // count

# --- LED 전체를 한 색으로 설정하는 함수 ---
def set_all_leds(r, g, b):
    """LED 10개를 모두 같은 색으로 설정"""
    for i in range(NUM_LEDS):
        np[i] = (r, g, b)
    np.write()

# --- 메인 루프 ---
print("=== 전자담배 검출 시스템 가동 ===")
print(f"주의 임계값: {THRESHOLD_WARN}")
print(f"경보 임계값: {THRESHOLD_ALERT}")

while True:
    value = read_average(gas_sensor)

    # 조건문으로 상태 판단
    if value >= THRESHOLD_ALERT:
        # 경보: 빨강
        set_all_leds(255, 0, 0)
        status = "경보"
    elif value >= THRESHOLD_WARN:
        # 주의: 노랑
        set_all_leds(255, 255, 0)
        status = "주의"
    else:
        # 안전: 초록
        set_all_leds(0, 255, 0)
        status = "안전"

    print(f"{status} | 센서값: {value}")
    time.sleep(0.5)
```

```python
# === 전자담배 검출 시스템 v2 (경보 애니메이션) ===
# 안전=초록 고정, 주의=노랑 숨쉬기, 경보=빨강 깜빡임

from machine import ADC, Pin
from neopixel import NeoPixel
import time

# --- 하드웨어 설정 ---
gas_sensor = ADC(Pin(26))
NUM_LEDS = 10
np = NeoPixel(Pin(16), NUM_LEDS)

# --- 임계값 설정 ---
THRESHOLD_WARN = 20000
THRESHOLD_ALERT = 35000

# --- 함수 모음 ---

def read_average(sensor, count=10):
    total = 0
    for i in range(count):
        total += sensor.read_u16()
        time.sleep_ms(10)
    return total // count

def set_all_leds(r, g, b):
    for i in range(NUM_LEDS):
        np[i] = (r, g, b)
    np.write()

def blink_red():
    """빨강 깜빡임 -- 위험 상황에서 주의를 끄는 효과"""
    set_all_leds(255, 0, 0)
    time.sleep(0.15)
    set_all_leds(0, 0, 0)
    time.sleep(0.15)

def breathe_yellow(step):
    """노랑 숨쉬기 -- 밝기가 서서히 변하는 효과
    step: 0~19 사이 값으로 호출할 때마다 증가"""
    # 삼각파로 밝기 변화: 0→255→0
    if step < 10:
        brightness = step * 25       # 0→250 (점점 밝게)
    else:
        brightness = (19 - step) * 25  # 250→0 (점점 어둡게)
    set_all_leds(brightness, brightness, 0)

# --- 메인 루프 ---
print("=== 전자담배 검출 시스템 v2 ===")

breath_step = 0  # 숨쉬기 애니메이션 단계

while True:
    value = read_average(gas_sensor)

    if value >= THRESHOLD_ALERT:
        # 경보: 깜빡임!
        blink_red()
        status = "경보"

    elif value >= THRESHOLD_WARN:
        # 주의: 숨쉬기 효과
        breathe_yellow(breath_step)
        breath_step = (breath_step + 1) % 20  # 0~19 순환
        status = "주의"
        time.sleep(0.1)

    else:
        # 안전: 초록 고정
        set_all_leds(0, 50, 0)  # 약간 어둡게 (눈 편하게)
        status = "안전"
        breath_step = 0  # 숨쉬기 단계 초기화
        time.sleep(0.5)

    print(f"{status} | 값: {value}")
```

선생님: "v1이랑 비교해보세요. 뭐가 달라졌나요?"

**[예상 Q&A]**

:::tip[% 20이 뭐예요?]
나머지 연산이에요. breath_step이 0,1,2,...19까지 가면 다시 0으로 돌아와요. 계속 순환하는 카운터를 만드는 트릭이에요.
:::

:::tip[안전할 때 (0,50,0)으로 바꾼 이유가 뭐예요?]
(0,255,0)은 너무 밝아서 계속 보면 눈이 피로해요. 평상시에는 은은하게, 경보 때만 화려하게. 실제 제품도 그렇게 디자인해요.
:::

:::tip[경보 때 blink_red()만 하고 sleep이 없는데요?]
blink_red() 안에 이미 sleep(0.15)가 두 번 들어있어요. 켜고 0.15초, 끄고 0.15초. 그래서 약 0.3초마다 깜빡여요.
:::

:::note[삼각파란?]
값이 올라갔다 내려갔다 하는 파형이에요. 0->250->0->250... 이렇게 반복해서 밝기가 서서히 변하는 '숨쉬기' 효과를 만드는 거예요.
:::
