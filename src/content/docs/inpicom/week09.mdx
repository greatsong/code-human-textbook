**[코드: step1_mapping.py]**
```python
# === 온도 → 색상 매핑 기초 ===
# map 함수로 값의 범위를 변환하기

from machine import I2C, Pin
from neopixel import NeoPixel
import bme280
import time

# --- 하드웨어 ---
i2c = I2C(0, sda=Pin(8), scl=Pin(9))
bme = bme280.BME280(i2c=i2c)
NUM_LEDS = 60
np = NeoPixel(Pin(16), NUM_LEDS)

# --- 핵심 함수: 값 범위 변환 ---
def map_value(value, in_min, in_max, out_min, out_max):
    """값을 한 범위에서 다른 범위로 변환
    예: map_value(25, 15, 35, 0, 255) → 127
    (15~35 범위의 25를 0~255 범위로 변환)"""
    value = max(in_min, min(in_max, value))  # 범위 제한
    return int((value - in_min) * (out_max - out_min) / (in_max - in_min) + out_min)

# --- 온도 → RGB 색상 변환 ---
def temp_to_color(temp_c):
    """온도를 파랑(추움)→빨강(더움)으로 변환
    15°C = 순수 파랑, 25°C = 초록, 35°C = 순수 빨강"""
    if temp_c <= 25:
        # 15~25°C: 파랑 → 초록
        r = 0
        g = map_value(temp_c, 15, 25, 0, 255)
        b = map_value(temp_c, 15, 25, 255, 0)
    else:
        # 25~35°C: 초록 → 빨강
        r = map_value(temp_c, 25, 35, 0, 255)
        g = map_value(temp_c, 25, 35, 255, 0)
        b = 0
    return (r, g, b)

# --- 전체 LED를 온도 색으로 ---
def set_all(color):
    for i in range(NUM_LEDS):
        np[i] = color
    np.write()

# --- 메인 루프 ---
print("=== 온도 시각화 조명 ===")

while True:
    temp_str = bme.values[0]       # "23.45C"
    temp_c = float(temp_str[:-1])  # 숫자만 추출

    color = temp_to_color(temp_c)
    set_all(color)

    print(f"온도: {temp_c:.1f}°C → RGB{color}")
    time.sleep(1)
```

#### 60 LED 막대 온도계 (15-30분)

**[코드: step2_bar_thermometer.py]**
```python
# === 60 LED 막대 온도계 ===
# 온도에 비례하여 LED가 채워지는 막대 그래프

from machine import I2C, Pin
from neopixel import NeoPixel
import bme280
import time

i2c = I2C(0, sda=Pin(8), scl=Pin(9))
bme = bme280.BME280(i2c=i2c)
NUM_LEDS = 60
np = NeoPixel(Pin(16), NUM_LEDS)

TEMP_MIN = 15.0  # 최소 표시 온도
TEMP_MAX = 35.0  # 최대 표시 온도

def map_value(value, in_min, in_max, out_min, out_max):
    value = max(in_min, min(in_max, value))
    return int((value - in_min) * (out_max - out_min) / (in_max - in_min) + out_min)

def update_bar(temp_c):
    """온도에 따라 LED 막대 그래프 표시"""
    # 온도 → 켜야 할 LED 개수
    num_on = map_value(temp_c, TEMP_MIN, TEMP_MAX, 0, NUM_LEDS)

    for i in range(NUM_LEDS):
        if i < num_on:
            # 위치에 따라 파랑→초록→빨강 그라데이션
            ratio = i / NUM_LEDS
            if ratio < 0.5:
                r = 0
                g = int(255 * ratio * 2)
                b = int(255 * (1 - ratio * 2))
            else:
                r = int(255 * (ratio - 0.5) * 2)
                g = int(255 * (1 - (ratio - 0.5) * 2))
                b = 0
            np[i] = (r, g, b)
        else:
            np[i] = (0, 0, 0)  # 나머지는 끔
    np.write()

# --- 메인 ---
print("=== LED 막대 온도계 ===")

while True:
    temp_str = bme.values[0]
    temp_c = float(temp_str[:-1])

    update_bar(temp_c)
    print(f"{temp_c:.1f}°C → LED {map_value(temp_c, TEMP_MIN, TEMP_MAX, 0, NUM_LEDS)}개 점등")
    time.sleep(0.5)
```

:::tip[float(temp_str[:-1])이 뭔 뜻이에요?]
`bme.values[0]`이 `'23.45C'`처럼 문자열로 오거든요. `[:-1]`로 마지막 글자(C)를 빼고, `float()`로 숫자로 바꾸는 거예요.

### 정리 (40-50분)

```
데이터 시각화의 핵심: 매핑(mapping)

  센서 데이터 ──→ [map 함수] ──→ 시각적 표현

  온도 → 색상 (오늘)
  CO2  → LED 개수 (7차시 bar)
  가스 → 깜빡임 속도 (3차시)
  소리 → 밝기 (응용 가능)

  같은 원리, 다른 매핑 = 다른 작품!
```

:::caution[정리 핵심]
**매핑(mapping)** 은 데이터 시각화의 가장 기본적인 개념입니다. 센서 데이터를 색상, 크기, 위치, 속도 등 다양한 시각적 요소로 변환할 수 있으며, `map_value` 함수는 이후 차시에서도 계속 사용됩니다.
:::
