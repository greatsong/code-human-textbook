**[코드: step1_multi_sensor.py]**
```python
# === 다중 센서 읽기 ===
# BME280 + SCD30 동시 데이터 수집

from machine import I2C, Pin
import bme280
import scd30
import time

i2c = I2C(0, sda=Pin(8), scl=Pin(9))

# --- I2C 스캔 ---
devices = i2c.scan()
print("I2C 장치:", [hex(d) for d in devices])
# 정상: [0x3c, 0x61, 0x76] 또는 [0x3c, 0x61, 0x77]

# --- 센서 초기화 ---
bme = bme280.BME280(i2c=i2c)
co2_sensor = scd30.SCD30(i2c, addr=0x61)
co2_sensor.start_continous_measurement()

print("=== 다중 센서 시작 ===")
time.sleep(2)  # SCD30 안정화

while True:
    # BME280 데이터
    bme_temp = bme.values[0]      # "23.4C"
    bme_hum = bme.values[2]       # "45.2%"
    bme_pres = bme.values[1]      # "1013.2hPa"

    # SCD30 데이터
    if co2_sensor.data_ready():
        co2, scd_temp, scd_hum = co2_sensor.read_measurement()

        print(f"[BME280] T:{bme_temp} H:{bme_hum} P:{bme_pres}")
        print(f"[SCD30]  CO2:{co2:.0f}ppm T:{scd_temp:.1f}C H:{scd_hum:.1f}%")
        print(f"[교차검증] 온도차: {abs(float(bme_temp[:-1])-scd_temp):.1f}C")
        print("---")

    time.sleep(2)
```

**[코드: step2_dashboard_ui.py]**
```python
# === 환경 대시보드 UI ===
# OLED 128x64에 모든 데이터를 보기 좋게 배치

from machine import I2C, Pin
import ssd1306
import bme280
import scd30
import time

i2c = I2C(0, sda=Pin(8), scl=Pin(9))
bme = bme280.BME280(i2c=i2c)
co2_s = scd30.SCD30(i2c, addr=0x61)
co2_s.start_continous_measurement()
oled = ssd1306.SSD1306_I2C(128, 64, i2c)

time.sleep(2)

# --- 환경 판정 ---
def judge_environment(co2, temp, hum):
    """6개 데이터로 종합 판정"""
    score = 0  # 0=최악, 100=최상

    # CO2 점수 (40점 만점)
    if co2 < 800:
        score += 40
    elif co2 < 1000:
        score += 30
    elif co2 < 1500:
        score += 15

    # 온도 점수 (30점 만점) - 20~26도가 쾌적
    if 20 <= temp <= 26:
        score += 30
    elif 18 <= temp <= 28:
        score += 20
    else:
        score += 5

    # 습도 점수 (30점 만점) - 40~60%가 쾌적
    if 40 <= hum <= 60:
        score += 30
    elif 30 <= hum <= 70:
        score += 20
    else:
        score += 5

    # 등급 판정
    if score >= 80:
        return score, "Excellent!"
    elif score >= 60:
        return score, "Good"
    elif score >= 40:
        return score, "Fair"
    else:
        return score, "Poor-Ventilate!"

# --- 메인 ---
print("=== 교실 환경 대시보드 ===")

while True:
    if co2_s.data_ready():
        co2, scd_t, scd_h = co2_s.read_measurement()
        bme_t = float(bme.values[0][:-1])
        bme_p = float(bme.values[1][:-3])

        # 평균 온습도 (두 센서 교차 검증)
        avg_temp = (bme_t + scd_t) / 2
        avg_hum = (float(bme.values[2][:-1]) + scd_h) / 2

        score, grade = judge_environment(co2, avg_temp, avg_hum)

        # OLED 표시
        oled.fill(0)

        # 상단: 등급 + 점수
        oled.text(f"{grade}", 0, 0)
        oled.text(f"{score}pt", 96, 0)
        oled.hline(0, 10, 128, 1)  # 구분선

        # 중단: 핵심 데이터
        oled.text(f"CO2:{co2:.0f}ppm", 0, 14)
        oled.text(f"T:{avg_temp:.1f}C", 0, 26)
        oled.text(f"H:{avg_hum:.0f}%", 64, 26)
        oled.text(f"P:{bme_p:.0f}hPa", 0, 38)

        # 하단: CO2 막대
        oled.hline(0, 50, 128, 1)
        bar_w = min(int(co2 / 2000 * 120), 120)
        oled.rect(4, 54, 120, 8, 1)
        oled.fill_rect(4, 54, bar_w, 8, 1)

        oled.show()

        print(f"[{grade} {score}pt] CO2:{co2:.0f} T:{avg_temp:.1f} H:{avg_hum:.0f}")

    time.sleep(2)
```

**[예상 Q&A]**

:::note[점수 기준은 누가 정한 거예요?]
임의로 교실 환경에 맞게 정한 거예요. 실제로는 환경부 실내 공기질 기준, WHO 가이드라인 등을 참고해야 해요. 기준을 바꾸면 결과도 달라지죠. '기준을 누가, 왜 정하는가'도 중요한 질문이에요.
:::

:::tip[두 센서의 온도 평균을 왜 내요?]
**센서 퓨전**이라는 기법이에요. 두 센서가 각각 약간씩 오차가 있지만, 평균 내면 오차가 줄어들어요. 자율주행차도 여러 센서의 데이터를 융합해서 판단합니다.
:::
