### 실습 1: 3장치 통합 (05-20분)

**[시각 자료: 연결도]**

```
Grove Shield

  [UART]      [I2C-1]      [I2C-2]
     ↑           ↑             ↑
  ┌──┴──────┐ ┌──┴──────┐  ┌──┴──┐
  │ Speech  │ │ BME280  │  │OLED │
  │Recognizr│ │ (0x76)  │  │(0x3C)│
  └─────────┘ └─────────┘  └─────┘

  UART: GP0(TX), GP1(RX) → 음성 명령
  I2C:  GP8(SDA), GP9(SCL) → 센서 + 디스플레이
```

**[코드: step1_voice_switch.py]**
```python
# === 음성 명령으로 데이터 전환 ===
# 명령에 따라 표시할 데이터 변경

from machine import UART, I2C, Pin
import bme280
import time

uart = UART(0, baudrate=9600, tx=Pin(0), rx=Pin(1))
i2c = I2C(0, sda=Pin(8), scl=Pin(9))
bme = bme280.BME280(i2c=i2c)

# --- 화면 모드 ---
MODE_ALL  = 0   # 전체 표시
MODE_TEMP = 1   # 온도 상세
MODE_HUM  = 2   # 습도 상세
MODE_PRES = 3   # 기압 상세

current_mode = MODE_ALL

# --- 명령 매핑 ---
# Grove Speech Recognizer 기본 명령어 활용
COMMAND_MAP = {
    1: (MODE_ALL,  "전체 보기"),    # "Turn on the light"
    2: (MODE_ALL,  "전체 보기"),    # "Turn off the light"
    7: (None,      "다음 화면"),     # "Up" → 다음 모드
    8: (None,      "이전 화면"),     # "Down" → 이전 모드
}

def next_mode():
    global current_mode
    current_mode = (current_mode + 1) % 4

def prev_mode():
    global current_mode
    current_mode = (current_mode - 1) % 4

# --- 메인 ---
print("=== 음성 제어 모니터 ===")
print("'Up' = 다음 화면, 'Down' = 이전 화면")

mode_names = ["ALL", "TEMP", "HUMIDITY", "PRESSURE"]

while True:
    # 음성 명령 확인
    if uart.any():
        data = uart.read()
        if data:
            cmd = data[0]
            if cmd == 7:
                next_mode()
                print(f"→ 다음: {mode_names[current_mode]}")
            elif cmd == 8:
                prev_mode()
                print(f"← 이전: {mode_names[current_mode]}")
            elif cmd in COMMAND_MAP:
                mode, name = COMMAND_MAP[cmd]
                if mode is not None:
                    current_mode = mode
                print(f"명령: {name}")

    # 센서 데이터
    temp = bme.values[0]
    hum = bme.values[2]
    pres = bme.values[1]

    # 현재 모드에 따라 출력
    if current_mode == MODE_ALL:
        print(f"[ALL] T:{temp} H:{hum} P:{pres}")
    elif current_mode == MODE_TEMP:
        print(f"[TEMP] {temp} (상세)")
    elif current_mode == MODE_HUM:
        print(f"[HUM] {hum} (상세)")
    elif current_mode == MODE_PRES:
        print(f"[PRES] {pres} (상세)")

    time.sleep(1)
```

**[코드: step2_voice_oled.py]**
```python
# === 음성 제어 OLED 모니터 ===
# 음성 명령으로 화면 모드 전환 + 실시간 갱신

from machine import UART, I2C, Pin
import ssd1306
import bme280
import time
import math

uart = UART(0, baudrate=9600, tx=Pin(0), rx=Pin(1))
i2c = I2C(0, sda=Pin(8), scl=Pin(9))
bme = bme280.BME280(i2c=i2c)
oled = ssd1306.SSD1306_I2C(128, 64, i2c)

current_mode = 0  # 0=all, 1=temp, 2=hum, 3=pres
mode_names = ["Overview", "Temperature", "Humidity", "Pressure"]

# --- 화면 그리기 함수 ---
def draw_overview(t, h, p):
    """전체 요약 화면"""
    oled.fill(0)
    oled.text("= Overview =", 15, 0)
    oled.hline(0, 10, 128, 1)
    oled.text(f"Temp: {t}", 0, 16)
    oled.text(f"Humi: {h}", 0, 30)
    oled.text(f"Pres: {p}", 0, 44)
    oled.text("Say 'Up'/'Down'", 0, 56)
    oled.show()

def draw_temp_detail(temp_c):
    """온도 상세 화면 + 아날로그 게이지"""
    oled.fill(0)
    oled.text("= Temperature =", 5, 0)
    oled.text(f"{temp_c:.1f} C", 35, 14)

    # 상태 판정
    if temp_c < 18:
        status = "Cold!"
    elif temp_c < 20:
        status = "Cool"
    elif temp_c <= 26:
        status = "Comfort"
    elif temp_c <= 28:
        status = "Warm"
    else:
        status = "Hot!"
    oled.text(status, 40, 28)

    # 막대 게이지 (15~35도 범위)
    bar_w = max(0, min(120, int((temp_c - 15) / 20 * 120)))
    oled.rect(4, 44, 120, 12, 1)
    oled.fill_rect(4, 44, bar_w, 12, 1)
    oled.text("15", 0, 58)
    oled.text("25", 55, 58)
    oled.text("35", 110, 58)
    oled.show()

def draw_hum_detail(hum_pct):
    """습도 상세 화면"""
    oled.fill(0)
    oled.text("= Humidity =", 15, 0)
    oled.text(f"{hum_pct:.0f} %", 45, 14)

    if hum_pct < 30:
        status = "Too Dry!"
    elif hum_pct < 40:
        status = "Dry"
    elif hum_pct <= 60:
        status = "Comfort"
    elif hum_pct <= 70:
        status = "Humid"
    else:
        status = "Too Humid!"
    oled.text(status, 35, 28)

    bar_w = max(0, min(120, int(hum_pct / 100 * 120)))
    oled.rect(4, 44, 120, 12, 1)
    oled.fill_rect(4, 44, bar_w, 12, 1)
    oled.text("0%", 0, 58)
    oled.text("50%", 50, 58)
    oled.text("100%", 96, 58)
    oled.show()

def draw_pres_detail(pres_hpa):
    """기압 상세 화면"""
    oled.fill(0)
    oled.text("= Pressure =", 12, 0)
    oled.text(f"{pres_hpa:.1f} hPa", 15, 14)

    # 기압으로 날씨 추정
    if pres_hpa > 1020:
        weather = "High P - Clear"
    elif pres_hpa > 1010:
        weather = "Normal"
    elif pres_hpa > 1000:
        weather = "Low P - Cloudy?"
    else:
        weather = "Storm coming?"
    oled.text(weather, 0, 30)

    # 고도 추정
    alt = 44330 * (1 - (pres_hpa / 1013.25) ** 0.1903)
    oled.text(f"Est.Alt: {alt:.0f}m", 0, 46)
    oled.show()

# --- 메인 루프 ---
print("=== 음성 제어 환경 모니터 ===")

while True:
    # 음성 명령 확인
    if uart.any():
        data = uart.read()
        if data:
            cmd = data[0]
            if cmd == 7:  # "Up"
                current_mode = (current_mode + 1) % 4
                print(f"→ {mode_names[current_mode]}")
            elif cmd == 8:  # "Down"
                current_mode = (current_mode - 1) % 4
                print(f"← {mode_names[current_mode]}")
            elif cmd == 1:  # "Turn on" → Overview
                current_mode = 0

    # 센서 읽기
    temp_s = bme.values[0]
    hum_s = bme.values[2]
    pres_s = bme.values[1]
    temp_c = float(temp_s[:-1])
    hum_f = float(hum_s[:-1])
    pres_f = float(pres_s[:-3])

    # 현재 모드에 따라 화면 그리기
    if current_mode == 0:
        draw_overview(temp_s, hum_s, pres_s)
    elif current_mode == 1:
        draw_temp_detail(temp_c)
    elif current_mode == 2:
        draw_hum_detail(hum_f)
    elif current_mode == 3:
        draw_pres_detail(pres_f)

    time.sleep(1)
```

**[예상 Q&A]**

:::note[Q: "화면 전환이 느릴 때가 있어요"]
음성 인식 -> UART 전송 -> 명령 처리 -> OLED 갱신까지 시간이 걸려요. 또 메인 루프의 `sleep(1)` 때문에 최대 1초 딜레이가 있을 수 있어요.
:::

:::note[Q: "더 많은 화면을 추가하고 싶어요"]
mode를 추가하고 draw 함수를 새로 만들면 돼요. 예를 들어 mode 4에 시계, mode 5에 그래프를 넣을 수 있어요.
:::
